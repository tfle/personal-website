## Pong

```javascript
// setup() runs once
function setup() {
  createCanvas(400, 400);
}

// Initialise variables
let xBall = 200;
let yBall = 200;
let xSpeed = 2;
let ySpeed = 5;
let scoreOne = 0;
let scoreTwo = 0;
let playerOne = 175;
let playerTwo = 175;

// draw() loops forever until stopped
function draw() {
  background("#FFFFFF");
  circle(xBall, yBall, 20); // ball position
  rect(200, 0, 1, 400); // line in the middle
  rect(30, playerOne, 10, 60);  // paddle 1
  rect(370, playerTwo, 10, 60); // paddle 2
  
  moveBall();
  bounceBall()
  keyPressed();
  paddleBounce();
  
  // display score
  textSize(16);
  text("Score " + scoreOne, 75, 30);
  text("Score " + scoreTwo, 275, 30);
}

// keyPressed() is called once every time a key is pressed
function keyPressed() {
  if (key == "w") {
    if (playerOne > 0) {
      playerOne -= 10;
    }
  } else if (key == "s") {
      if (playerOne < 340) {
        playerOne += 10; 
      }
  } else if (keyCode == UP_ARROW) {
      if (playerTwo > 0) {
        playerTwo -= 10;    
      }
  } else if (keyCode == DOWN_ARROW) {
      if (playerTwo < 340) {
        playerTwo += 10;
      }
  }
}

function moveBall() {
  xBall += xSpeed;
  yBall += ySpeed;
}

function bounceBall() {
  if (xBall < 10){
    scoreTwo++;
    resetBall()
  }  
  if (xBall > 390) {
    scoreOne++;
    resetBall()
  }
  if (yBall < 10 || yBall > 390) {
    ySpeed *= -1;
  }
}

function resetBall() {
  xBall = 200;
  yBall= 200;
}

function paddleBounce() {
  // left paddle
  if (xBall < 45 && (yBall > playerOne) && (yBall < playerOne + 60)) {
    xSpeed *= -1.1;
  }
  // right paddle
  if (xBall > 355 && (yBall > playerTwo) && (yBall < playerTwo + 60)) {
    xSpeed += -1.1;
  }
}
```
